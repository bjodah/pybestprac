[docs]def allclose(a, b, rtol=1e-8, atol=1e-8):
    if len(a) != len(b):
        raise ValueError("Incomaptible lengths")
    for idx, (x, y) in enumerate(zip(a, b)):
        if abs(x-y) > atol+rtol*abs(y):
            return idx
    return -1
